#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    int arr[n];
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    for (int i = 0; i < n - 1; i++) {
        for (int j = i + 1; j < n; j++) {
            if (arr[j] < arr[i]) {
                int temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }

    int oneArray = arr[0];
    int count = 1;

    for (int i = 1; i < n; i++) {
        if (arr[i] == oneArray) {
            count++;
        } else {
            cout << oneArray << " aparece " << count << " vez(es)\n";
            oneArray = arr[i];
            count = 1;
        }
    }
    cout << oneArray << " aparece " << count << " vez(es)\n";

    return 0;
}
